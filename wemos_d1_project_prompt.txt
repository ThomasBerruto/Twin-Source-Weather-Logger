// -----------------------------
// WEATHER FETCH
// -----------------------------
bool fetchWeather(String zipcode, float &tempF, float &feelsF, float &humidity) {
  if (WiFi.status() != WL_CONNECTED) return false;

  HTTPClient http;
  //HttpClient http(c);
  String url = "http://wttr.in/" + zipcode + "?format=j1";
  http.begin(url);

  int code = http.GET();
  if (code != 200) {
    http.end();
    return false;
  }

  String payload = http.getString();
  http.end();

  StaticJsonDocument<4096> doc;
  if (deserializeJson(doc, payload)) return false;

  tempF = doc["current_condition"][0]["temp_F"].as<float>();
  feelsF = doc["current_condition"][0]["FeelsLikeF"].as<float>();
  humidity = doc["current_condition"][0]["humidity"].as<float>();

  return true;
}

// -----------------------------
// SETUP
// -----------------------------
void setup() {
  Serial.begin(115200);

  dht.begin();

  // WiFi
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
  
  // Set timezone and NTP servers
  configTime(-6 * 3600, 3600, "pool.ntp.org", "time.nist.gov");
  //  -6*3600 = CST offset
  //   3600   = daylight savings offset
  Serial.println("Syncing time with NTP...");
  
  struct tm timeinfo;
  while (!getLocalTime(&timeinfo)) {
    Serial.println("Waiting for NTP...");
    delay(500);
  }
  
  Serial.println("Time synced!");

  // SD card
  if (!SD.begin(SD_CS_PIN)) {
    Serial.println("SD init failed!");
  } else {
    Serial.println("SD OK");
  }

  // RTC
  if (!rtc.begin()) {
    Serial.println("RTC not found!");
  }
  if (!rtc.isrunning()) {
    Serial.println("RTC not running, setting time...");
    rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
  }

  // Create CSV if not exists
  if (!SD.exists("/sensor_log.csv")) {
    File f = SD.open("/sensor_log.csv", FILE_WRITE);
    f.println("date_time,date,hour,minutes,seconds,sensor_name,measurement,sensor_value,Units");
    f.close();
  }

  // IoT Cloud
  ArduinoCloud.setBoardId(DEVICE_ID);
  ArduinoCloud.setSecretDeviceKey(SECRET_KEY);

  ArduinoCloud.addProperty(temperature, READWRITE);
  ArduinoCloud.addProperty(humidity, READWRITE);
  ArduinoCloud.addProperty(webTemp, READWRITE);
  ArduinoCloud.addProperty(webHumidity, READWRITE);

  ArduinoCloud.begin(ArduinoIoTPreferredConnection);
}
